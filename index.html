    <script>

        const API_KEY = 'e6f43714ba7e4cb9b53055d0869b3e59';

        const BASE_URL = 'https://api.openweathermap.org/data/2.5/weather';

        let favorites = JSON.parse(localStorage.getItem('weatherFavorites')) || [];


        async function fetchWeather(city) {

            try {

                const container = document.getElementById('weatherContainer');

                container.innerHTML = '<div class="loading">⏳ กำลังโหลด...</div>';


                const response = await fetch(`${BASE_URL}?q=${city}&appid=${API_KEY}&units=metric&lang=th`);

                if (!response.ok) throw new Error('เมืองไม่พบ');


                const data = await response.json();

                displayWeather(data);

                document.getElementById('searchInput').value = '';

            } catch (error) {

                document.getElementById('weatherContainer').innerHTML = `<div class="error">❌ ${error.message}</div>`;

            }

        }


        function displayWeather(data) {

            const { name, main, weather, wind, clouds, sys } = data;

            const iconMap = {

                'clear sky': '☀️',

                'few clouds': '🌤️',

                'scattered clouds': '☁️',

                'broken clouds': '☁️',

                'shower rain': '🌧️',

                'rain': '🌧️',

                'thunderstorm': '⛈️',

                'snow': '❄️',

                'mist': '🌫️'

            };


            const weatherDesc = weather[0].description.toLowerCase();

            const icon = iconMap[weatherDesc] || '🌡️';

            const isFavorite = favorites.includes(name);


            const html = `

                <div class="weather-card">

                    <div class="city-info">

                        <div class="city-name">${name}, ${data.sys.country}</div>

                        <button class="btn-favorite" onclick="toggleFavorite('${name}')">

                            ${isFavorite ? '❤️' : '🤍'}

                        </button>

                    </div>


                    <div class="weather-main">

                        <div class="temp-section">

                            <div class="temperature">${Math.round(main.temp)}°C</div>

                            <div class="weather-description">${weather[0].description}</div>

                        </div>

                        <div class="temp-section">

                            <div class="weather-icon">${icon}</div>

                        </div>

                    </div>


                    <div class="weather-details">

                        <div class="detail-item">

                            <div class="detail-label">ความรู้สึก</div>

                            <div class="detail-value">${Math.round(main.feels_like)}°C</div>

                        </div>

                        <div class="detail-item">

                            <div class="detail-label">ความชื้น</div>

                            <div class="detail-value">${main.humidity}%</div>

                        </div>

                        <div class="detail-item">

                            <div class="detail-label">ความดัน</div>

                            <div class="detail-value">${main.pressure} hPa</div>

                        </div>

                        <div class="detail-item">

                            <div class="detail-label">ลมพัด</div>

                            <div class="detail-value">${Math.round(wind.speed)} m/s</div>

                        </div>

                    </div>

                </div>

            `;

            document.getElementById('weatherContainer').innerHTML = html;

        }


        function searchWeather() {

            const city = document.getElementById('searchInput').value.trim();

            if (city) fetchWeather(city);

        }


        function handleKeyPress(event) {

            if (event.key === 'Enter') searchWeather();

        }


        function getLocation() {

            if (navigator.geolocation) {

                document.getElementById('weatherContainer').innerHTML = '<div class="loading">📍 กำลังหาตำแหน่ง...</div>';

                navigator.geolocation.getCurrentPosition(

                    (position) => {

                        fetchWeatherByCoords(position.coords.latitude, position.coords.longitude);

                    },

                    () => {

                        document.getElementById('weatherContainer').innerHTML = '<div class="error">❌ ไม่สามารถหาตำแหน่ง</div>';

                    }

                );

            }

        }


        async function fetchWeatherByCoords(lat, lon) {

            try {

                const response = await fetch(`${BASE_URL}?lat=${lat}&lon=${lon}&appid=${API_KEY}&units=metric&lang=th`);

                const data = await response.json();

                displayWeather(data);

            } catch (error) {

                document.getElementById('weatherContainer').innerHTML = '<div class="error">❌ เกิดข้อผิดพลาด</div>';

            }

        }


        function toggleFavorite(city) {

            if (favorites.includes(city)) {

                favorites = favorites.filter(c => c !== city);

            } else {

                favorites.push(city);

            }

            localStorage.setItem('weatherFavorites', JSON.stringify(favorites));

            updateFavorites();

        }


        function updateFavorites() {

            const listContainer = document.getElementById('favoritesList');

            if (favorites.length === 0) {

                listContainer.innerHTML = '<p style="color: white;">ยังไม่มีเมืองโปรด</p>';

                return;

            }

            listContainer.innerHTML = favorites.map(city => `

                <div style="background: rgba(255,255,255,0.9); padding: 10px; border-radius: 8px;">

                    <button class="favorite-btn" style="width: 100%; margin-bottom: 5px;" onclick="fetchWeather('${city}')">${city}</button>

                    <button class="delete-favorite" onclick="removeFavorite('${city}')">❌ ลบ</button>

                </div>

            `).join('');

        }


        function removeFavorite(city) {

            favorites = favorites.filter(c => c !== city);

            localStorage.setItem('weatherFavorites', JSON.stringify(favorites));

            updateFavorites();

        }


        updateFavorites();

        fetchWeather('Bangkok');
